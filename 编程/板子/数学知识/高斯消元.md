#include<iostream>

#include<cmath>

using namespace std;

double x[200][200];

double eps=1e-9;

double ans[200];

int main(){

    ios::sync_with_stdio(false);

    cin.tie(0);cout.tie(0);

    int n;cin>>n;

    for(int i=1;i<=n;i++){

        for(int j=1;j<=n+1;j++){

            cin>>x[i][j];

        }

    }

    for(int i=1;i<n;i++){//遍历每一位

        int t=i;

        for(int j=i;j<=n;j++){//找到当前最大位

            if(x[i][j]>x[i][t]){

                t=j;

            }

        }//x[列][行]

        if(fabs(x[i][t])<eps){

            cout<<"No Solution";

            return 0;

        }

        for(int j=1;j<=n+1;j++){

            swap(x[j][i],x[j][t]);

        }

        for(int j=i+1;j<=n;j++){

            double bs=x[i][j]/x[i][i];

            for(int k=i;k<=n+1;k++){

                x[k][j]-=bs*x[k][i];

            }

        }

    }

    ans[n]=x[n+1][n];

    for(int i=n-1;i>=1;i--){

        ans[i]=x[n+1][i];

        for(int j=n;j>i;j--){

            ans[i]-=x[j][i]*ans[j];

        }

    }

    for(int i=1;i<=n;i++){

        printf("%.2lf\n",ans[i]);

    }

    return 0;

}
