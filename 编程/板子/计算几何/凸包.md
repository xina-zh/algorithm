#include<iostream>
#include<cmath>
#include<algorithm>
#include<vector>

using namespace std;
typedef pair<double,double> pdd;
#define x first
#define y second
const int N=1e5+10;

int sta[N],top;
int used[N];
pdd operator-(pdd a,pdd b){
    return {a.x-b.x,a.y-b.y};
}

double operator*(pdd a,pdd b){
    return a.x*b.y-a.y*b.x;
}
double cross(pdd a,pdd b,pdd c){
    return (b-a)*(c-a);
}
vector<pdd> a;
double jl(pdd a,pdd b){
    double dx=a.x-b.x;
    double dy=a.y-b.y;
    return sqrtl(dx*dx+dy*dy);
}

  

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    int n;cin>>n;
    for(int i=1;i<=n;i++){
        double j,k;cin>>j>>k;
        a.push_back({j,k});
    }
    sort(a.begin(),a.end());
    a.erase(unique(a.begin(),a.end()),a.end());
    int len=a.size();
    for(int i=0;i<len;i++){
        while(top>=2&&cross(a[sta[top-1]],a[sta[top]],a[i])<0){
            used[sta[top]]=0;
            top--;
        }
        sta[++top]=i;
        used[i]=1;
    }
    used[0]=0;
    for(int i=len-2;i>=0;i--){
        if(used[i]) continue;
        while(top>=2&&cross(a[sta[top-1]],a[sta[top]],a[i])<0){
            used[sta[top]]=0;
            top--;
        }
        sta[++top]=i;
        used[i]=1;
    }
    double ans=0;
    for(int i=2;i<=top;i++){
        ans+=jl(a[sta[i-1]],a[sta[i]]);
    }
    printf("%.2lf",ans);
    return 0;
}
